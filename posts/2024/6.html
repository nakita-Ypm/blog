Zod Validator ​

ts
import { Hono } from 'hono'
import { BunAdapter } from './adapter/bun_adapter'
import { z } from 'zod'
import { zValidator } from '@hono/zod-validator'

const app = new Hono()

const schema = z.object({
  name: z.string(),
  age: z.number(),
})

app.post('/author', zValidator('json', schema), (c) => {
  const data = c.req.valid('json')
  console.log(data)
  return c.json({
    success: true,
    message: `${data.name} is ${data.age}`,
  })
})

const serve = BunAdapter.serveApp(app)

export default serve

if (process.pid) {
  console.log('This process is server pid ' + process.pid)
}

Hook ​

ts
app.post(
  '/post',
  zValidator('json', schema, (result, c) => {
    if (!result.success) {
      return c.json({
        success: false,
        message: 'Invalid!',
      })
    }
    return c.json({
      success: true,
      message: 'Valid!',
    })
  }),
)
