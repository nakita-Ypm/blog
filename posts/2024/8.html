ts
// index.ts
import { createRoute, OpenAPIHono, z } from '@hono/zod-openapi'
import { swaggerUI } from '@hono/swagger-ui'
import { basicAuth } from 'hono/basic-auth'
import authors from './handler/authors'
import books from './handler/books'

const app = new OpenAPIHono()

export const honoSchema = z.object({
  message: z.string().openapi({
    example: 'HonoğŸ”¥',
  }),
})

const route = createRoute({
  method: 'get',
  path: '/hono',
  responses: {
    200: {
      content: {
        'application/json': {
          schema: honoSchema,
        },
      },
      description: 'HonoğŸ”¥',
    },
  },
  tags: ['Hono'],
})

app.openapi(route, (c) => {
  return c.json({ message: 'HonoğŸ”¥' })
})

app.use('/auth/*', basicAuth({ username: 'hono', password: 'swagger' }))

app.get('/auth/ui', swaggerUI({ url: '/auth/doc' }))

app.doc('/auth/doc', {
  info: {
    title: 'Hono API',
    version: 'v1',
  },
  openapi: '3.1.0',
})

app.route('/', authors)
app.route('/', books)

export default app
