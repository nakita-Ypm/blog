  • Zod OpenAPI

  • Swagger UI

Hono API OAS ​

 Honoの機能を使用し、APIから、ドキュメントを自動生成。

OpenAPIのRoute定義を追加 ​

 スキーマは別のファイルに定義している。

Schema定義
ts
import { z } from '@hono/zod-openapi'

export const MessageSchema = z.object({
  message: z.string().openapi({
    example: 'Hono🔥',
  }),
})

export const ParamsSchema = z.object({
  id: z
    .string()
    .min(3)
    .openapi({
      param: {
        name: 'id',
        in: 'path',
      },
      example: '1212121',
    }),
})

export const UserSchema = z
  .object({
    id: z.string().openapi({
      example: '123',
    }),
    name: z.string().openapi({
      example: 'John Doe',
    }),
    age: z.number().openapi({
      example: 42,
    }),
  })
  .openapi('User')
ts
import { createRoute } from '@hono/zod-openapi'
import { MessageSchema, ParamsSchema, UserSchema } from '../schemas'

// hono
export const hono = createRoute({
  method: 'get',
  path: '/hono',
  responses: {
    200: {
      content: {
        'application/json': {
          schema: MessageSchema,
        },
      },
      description: 'Hono🔥',
    },
  },
})

// user
export const user = createRoute({
  method: 'get',
  path: '/users/{id}',
  request: {
    params: ParamsSchema,
  },
  responses: {
    200: {
      content: {
        'application/json': {
          schema: UserSchema,
        },
      },
      description: 'Retrieve the user',
    },
  },
})

OpenAPIHono ​

 OpenAPIHonoを用いて、簡単なAPIを作成

ts
import { OpenAPIHono } from '@hono/zod-openapi'
import { hono } from '../../api/src'

export class HonoHandler {
  static apply(app: OpenAPIHono) {
    return app.openapi(hono, (c) => {
      return c.json({ message: 'Hono🔥' })
    })
  }
}
ts
import { OpenAPIHono } from '@hono/zod-openapi'
import { user } from '../../api/src'

export class UserHandler {
  static apply(app: OpenAPIHono) {
    return app.openapi(user, (c) => {
      const { id } = c.req.valid('param')
      return c.json({
        id,
        age: 20,
        name: 'Ultra-man',
      })
    })
  }
}

Swagger ​

 OASドキュメントを作成する。機密情報の場合認証をするのが適切。

ts
import { OpenAPIHono } from '@hono/zod-openapi'
import { swaggerUI } from '@hono/swagger-ui'

export class SwaggerHandler {
  static apply(app: OpenAPIHono) {
    return app
      .doc('/doc', {
        openapi: '3.0.0',
        info: {
          version: '1.0.0',
          title: 'Hono API🔥',
        },
      })
      .get('/ui', swaggerUI({ url: '/doc' }))
  }
}

 自身のローカルホストhttp://localhost:***/uiにアクセスすると、OASドキュメントが表示される。
